import turtle

# Mr. Sharp, please put a minimum of 6 and a maximum of 15 icons to see the full product full-screen.

screen = turtle.Screen()  # Needed for text input

tina = turtle.Turtle()
tina.shape('turtle')
tina.speed(10)

num = screen.textinput(
    "Number of logos",
    "How many Netflix logos would you like to draw?"
)

num = int(num)

tina.penup()
tina.goto(-500, 120)
tina.pendown()

for n in range(num):
    tina.begin_fill()
    tina.color('black')
    for i in range(4):
        tina.forward(10)
        tina.circle(90, 90)
        tina.forward(50)
    tina.end_fill()

    tina.penup()
    tina.right(90)
    tina.forward(-75)
    tina.left(90)
    tina.backward(55)
    tina.pendown()

    for i in range(2):
        tina.color('red')
        tina.begin_fill()
        tina.forward(20)
        tina.left(90)
        tina.forward(80)
        tina.left(90)
        tina.end_fill()

    tina.penup()
    tina.forward(50)
    tina.pendown()
    tina.begin_fill()
    for i in range(2):
        tina.color('red')
        tina.forward(20)
        tina.left(90)
        tina.forward(80)
        tina.left(90)
    tina.end_fill()

    tina.penup()
    tina.forward(10)
    tina.left(130)
    tina.pendown()
    tina.begin_fill()
    tina.color('dark red')
    for i in range(2):
        tina.forward(95)
        tina.right(90)
        tina.forward(15)
        tina.right(90)
    tina.end_fill()

    tina.penup()
    tina.right(220)
    tina.forward(70)
    tina.left(90)
    tina.forward(250)
    tina.pendown()

    if n == 4:
        tina.penup()
        tina.goto(-500, -130)
        tina.pendown()

    if n == 9:
        tina.penup()
        tina.goto(-500, -380)
        tina.pendown()
